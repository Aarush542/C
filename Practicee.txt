#include<stdio.h>
int main()
{
    int a,n;
    printf("Enter a number a");
    scanf("%d",&a);
    printf("Enter a number n");
    scanf("%d",&n);
    while(a<=n)
    {
        printf("%d\n",a);
        a++;
    }
    return 0;
}
